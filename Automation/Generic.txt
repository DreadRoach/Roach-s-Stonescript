//Generic automation script for universal & basic use cases.
//To use, place above all other automation/farming scripts.

//Imports the Mindstone Button, needed
//to edit scripts mid-level on mobile.
import UI/MindstoneButton

//Default healing script.
?hp < 7
 activate potion

//Sets the default equipment.
equipR ourobouros
equipL triskelion

//Changes equipment for the right hand.
?foe.distance < 9999
 ?foe.distance <= 16 & foe.distance >= 11
  ?foe.armor >= 1 | foe = immune_to_ranged
   equipR dash
 :?foe.distance =< 5 & foe.armor => 1
   equip heavy hammer
 :
  equipR shield ah

//Changes equipment for the left hand.
?pickup.distance <= 20 & pickup.distance < foe.distance
 equipL star
:?foe.distance <= 5 & foe.armor <= 0
^| foe = immune_to_ranged & foe.armor <= 0
 equipL sword dL
:?foe.distance <= 20 & foe ! magic_resist
 equipL wand dL
:?foe.distance <= 22
 equipL cross
